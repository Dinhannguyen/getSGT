-- Blox Fruits Auto-Hop Script with Blacklist and GUI (Optimized)

-- Services
local Players = game:GetService("Players")
local TeleportService = game:GetService("TeleportService")
local HttpService = game:GetService("HttpService")
local RunService = game:GetService("RunService")
local UserInputService = game:GetService("UserInputService")
local ReplicatedStorage = game:GetService("ReplicatedStorage")

-- Local Player
local player = Players.LocalPlayer
local username = player.Name
local blacklistFile = "blacklist.json"

-- Trạng thái Hopping
local isSearching = false
local isHopping = false

-- Blacklist Management
local function loadBlacklist()
    if isfile(blacklistFile) then
        local content = readfile(blacklistFile)
        return HttpService:JSONDecode(content) or {}
    else
        return {}
    end
end

local function saveBlacklist(blacklist)
    writefile(blacklistFile, HttpService:JSONEncode(blacklist))
end

-- Add own username to blacklist if not already present
local blacklist = loadBlacklist()
if not table.find(blacklist, username) then
    table.insert(blacklist, username)
    saveBlacklist(blacklist)
end

-- ==========================================
-- CORE HOPPING LOGIC (Tối ưu theo chuẩn mới)
-- ==========================================

-- Hàm thực thi Teleport qua __ServerBrowser kết hợp Anti-Stuck
local function executeTeleport(jobId)
    if isHopping then return end
    isHopping = true
    
    local initialJobId = game.JobId
    local remote = ReplicatedStorage:FindFirstChild("__ServerBrowser")
    
    if remote then
        pcall(function()
            remote:InvokeServer("teleport", jobId)
        end)
    else
        -- Fallback nếu game lỡ xóa mất remote __ServerBrowser
        pcall(function()
            TeleportService:TeleportToPlaceInstance(game.PlaceId, jobId, player)
        end)
    end

    -- Anti-Stuck: Kiểm tra sau 10 giây, nếu chưa nhảy được thì reset để hop tiếp
    task.delay(10, function()
        if game.JobId == initialJobId then
            isHopping = false -- Hủy trạng thái đang hop để 2 luồng quét tiếp tục
        end
    end)
end

-- Luồng quét bằng URL
local function hopViaURL()
    pcall(function()
        local serversUrl = "https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"
        local response = game:HttpGet(serversUrl)
        local servers = HttpService:JSONDecode(response).data
        
        local availableServers = {}
        for _, server in ipairs(servers) do
            if server.playing < server.maxPlayers and server.id ~= game.JobId then
                table.insert(availableServers, server.id)
            end
        end
        
        if #availableServers > 0 then
            local randomJobId = availableServers[math.random(1, #availableServers)]
            executeTeleport(randomJobId)
        end
    end)
end

-- Luồng quét bằng Server Browser ẩn của Blox Fruits
local function hopViaServerBrowser()
    local remote = ReplicatedStorage:FindFirstChild("__ServerBrowser")
    if remote then
        pcall(function()
            local response = remote:InvokeServer(1, "Singapore") -- Có thể đổi region tùy ý
            if type(response) == "table" then
                local availableServers = {}
                for jobId, info in pairs(response) do
                    if type(info) == "table" then
                        local pCount = tonumber(info.Count) or 0
                        -- Lọc server có người chơi nhưng chưa đầy (Blox Fruits max 12)
                        if pCount > 0 and pCount < 12 and jobId ~= game.JobId then 
                            table.insert(availableServers, jobId)
                        end
                    end
                end
                
                if #availableServers > 0 then
                    local randomJobId = availableServers[math.random(1, #availableServers)]
                    executeTeleport(randomJobId)
                end
            end
        end)
    end
end

-- Hàm quản lý chạy song song 2 luồng
local function hopServer()
    if isSearching then return end
    isSearching = true
    
    -- Luồng 1: Chạy quét URL
    task.spawn(function()
        while isSearching do
            if not isHopping then
                hopViaURL()
            end
            task.wait(3) -- Delay chống spam request
        end
    end)
    
    -- Luồng 2: Chạy quét Remote
    task.spawn(function()
        while isSearching do
            if not isHopping then
                hopViaServerBrowser()
            end
            task.wait(3.5) -- Lệch nhịp một chút so với URL để tối ưu
        end
    end)
end

-- ==========================================
-- GUI SETUP & MAIN LOOPS (Phần cũ của bạn)
-- ==========================================

local gui = Instance.new("ScreenGui")
gui.Name = "AutoHopGUI"
gui.ResetOnSpawn = false
gui.Parent = player:WaitForChild("PlayerGui")

local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 300, 0, 50)
frame.Position = UDim2.new(0.5, 0, 1, -10)
frame.AnchorPoint = Vector2.new(0.5, 1)
frame.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
frame.BackgroundTransparency = 0.5
frame.BorderSizePixel = 0
frame.Parent = gui

local statusLabel = Instance.new("TextLabel")
statusLabel.Size = UDim2.new(1, 0, 1, 0)
statusLabel.BackgroundTransparency = 1
statusLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
statusLabel.TextSize = 18
statusLabel.Font = Enum.Font.SourceSansBold
statusLabel.Text = "Auto-Hop Active"
statusLabel.Parent = frame

local function updateStatus(text)
    statusLabel.Text = text
end

local idleTime = 0
local noSpawnTime = 0
local lastPosition = nil
local checkInterval = 1  
local blacklistChecked = false

-- Main Loop
task.spawn(function()
    while true do
        task.wait(checkInterval)
        
        -- Blacklist Check
        if not blacklistChecked and player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            blacklist = loadBlacklist()
            for _, blUsername in ipairs(blacklist) do
                if blUsername ~= username and Players:FindFirstChild(blUsername) then
                    for i = 10, 1, -1 do
                        updateStatus("Blacklisted player detected! Hopping in " .. i .. "s")
                        task.wait(1)
                    end
                    hopServer()
                    break
                end
            end
            blacklistChecked = true
        end

        -- Check No Spawn
        if not player.Character then
            noSpawnTime = noSpawnTime + checkInterval
            if noSpawnTime > 120 then
                updateStatus("No spawn detected! Hopping...")
                hopServer()
            end
        else
            noSpawnTime = 0
            -- Check Idle
            local humanoidRootPart = player.Character:FindFirstChild("HumanoidRootPart")
            if humanoidRootPart then
                local currentPosition = humanoidRootPart.Position
                if lastPosition and (currentPosition - lastPosition).Magnitude < 1 then
                    idleTime = idleTime + checkInterval
                    if idleTime > 120 then
                        updateStatus("Idle detected! Hopping...")
                        hopServer()
                    end
                else
                    idleTime = 0
                end
                lastPosition = currentPosition
            end
        end

        -- Reset GUI text
        if statusLabel.Text ~= "Auto-Hop Active" and not isSearching then
            task.wait(2)
            updateStatus("Auto-Hop Active")
        elseif isSearching then
            updateStatus("Đang tìm Server để Hop...")
        end
    end
end)

-- Anti-AFK

    while true do
        task.wait(60)  
        local vu = game:GetService("VirtualUser")
        vu:CaptureController()
        vu:ClickButton2(Vector2.new())
    end
end)
